void print(int* m, int size)
{
	for (int i = 0; i < size; i++)
	{
		cout << m[i] << "  "; 
	}
	cout << endl;
}
void fill(int* m, int size)
{
	for (int i = 0; i < size; i++)
	{
		m[i] = rand() % 50;
	}
}
void sort(int*m, int size,bool f)
{
	
	if (f == true)
	{
		for (int i = 0; i < size - 1; i++)
		{
			for (int j = 0; j < size - 1; j++)
			{
				if (m[j] > m[j + 1])
				{
					swap(m[j], m[j + 1]);
				}
			}
		}
	}
	else
	{
		for (int i = 0; i < size - 1; i++)
		{
			for (int j = 0; j < size - 1; j++)
			{
				if (m[j] < m[j + 1])
				{
					swap(m[j], m[j + 1]);
				}
			}
		}
	}
}
int maximum(int*m, int size)
{
	int max=0;
	for (int i = 0; i < size; i++)
	{
		if (m[i] > max)
		{
			max = m[i];
		}
	}
	cout << max << endl;
	return max;
}
